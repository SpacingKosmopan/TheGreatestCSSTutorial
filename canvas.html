const canvas = document.getElementById('myCanvas');
const ctx = canvas.getContext('2d'); - getting element in JS

Basic shapes

Rectangle

  • ctx.fillRect(x, y, width, height) - drawing rectangle
  • ctx.strokeRect(x, y, width, height) - drawing rectangle border
  • ctx.clearRect(x, y, width, height) - rectangle rubber

Lines, triangles

  • ctx.beginPath() - begin (at the beginning)
  • ctx.moveTo(x, y) - moves pencil to location
  • ctx.lineTo(x, y) - draws line from current point to new point
  • ctx.closePath() - last point to start point

Circles, arcs

ctx.arc(x, y, radius, startAngle, endAngle);
Full circle: 

ctx.beginPath();
// Middle (200, 200), radius 50, start from 0, ends on full circle
ctx.arc(200, 200, 50, 0, Math.PI * 2); 
ctx.fillStyle = 'purple';
ctx.fill();

Styling

  • ctx.fillStyle = 'red';
  • ctx.strokeStyle = 'blue';
  • ctx.lineWidth = 5;

Drawing frames

requestAnimationFrame(funtionName); - the same as Update() in Unity 

let x = 0; // Starting pos

function animate() {
    ctx.clearRect(0, 0, canvas.width, canvas.height); // Clear the screen
    
    ctx.beginPath();
    ctx.arc(x, 100, 20, 0, Math.PI * 2); // circle drawing
    ctx.fillStyle = 'orange';
    ctx.fill();
    
    x += 2; // Position change
    
    if (x > canvas.width) x = 0; // Reset

    requestAnimationFrame(animate); // Animation loop
}

animate(); // Loop startup

ctx.drawImage();

ctx.drawImage(img, sx, sy, sWidth, sHeight, dx, dy, dWidth, dHeight);
Mnemotechnika: S like Source, D like Destination.

  • img: Image object (atlas).
  • sx, sy: Sprite in atlas position (X and Y in image file).
  • sWidth, sHeight: Sprite size in atlas.
  • dx, dy: Drawing position (X and Y on screen).
  • dWidth, dHeight: Image size on screen (scaling).

Spritesheet data Loading

let atlasData = null;
let atlasImage = new Image();
async function initGame(){
  try {
    const response = await fetch('./spritesheet.json');
    atlasData = await response.json();

    atlasImage.src = './' + atlasData.meta.image;

    atlasImage.onload = () => {
      console.log("Startup...");
      startGame();
    };
  } catch (errors) {
    console.error("Errors: ", errors)
  }
}

initGame();

Sprites drawing help function

function drawSprite(spriteName, dX, dY){
  const sprite = atlasData.frames[spriteName];

  if(!sprite) {
    console.error(`${spriteName} not found`);
    return;
  }

  ctx.drawImage(
    atlasImage,
    sprite.frame.x, sprite.frame.y,
    sprite.frame.w, sprite.frame.h,
    dX, dY, sprite.frame.w, sprite.frame.h
  );
}

Input Checking

const klawisze = {
    ArrowUp: false,
    ArrowDown: false,
    ArrowLeft: false,
    ArrowRight: false,
    w: false,
    s: false,
    a: false,
    d: false
};

window.addEventListener('keydown', (event) => {
    if (event.key in klawisze) {
        klawisze[event.key] = true;
    }
});

window.addEventListener('keyup', (event) => {
    if (event.key in klawisze) {
        klawisze[event.key] = false;
    }
});
